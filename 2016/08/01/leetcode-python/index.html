def searchRange(nums, target):
lo, hi = 0, len(nums)-1
while lo <= hi:
mid = (hi + lo)//2
if nums[mid] == target:
while nums[lo] != target:
if nums[lo] < target:
lo += 1
while nums[hi] != target:
if nums[hi] > target:
hi -= 1
return [lo, hi]
elif nums[mid] < target:
lo = mid + 1
elif nums[mid] > target:
hi = mid - 1
return [-1, -1]

start = nums.index(target)

ncount = nums.count(target)

res = [start, start + ncount - 1]

return res
